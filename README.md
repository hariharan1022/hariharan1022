## <img src="https://media.giphy.com/media/KzJkzjggfGN5Py6nkT/giphy.gif" width="28"> &nbsp;React Component Showcase

```tsx
// ── Skyrovix ProductCard.tsx  ·  How I think in React + TypeScript ─────────

import { useState } from "react";
import { motion, AnimatePresence } from "framer-motion";

interface ProductCardProps {
  title       : string;
  price       : number;
  platform    : "etsy" | "ebay" | "pod" | "custom";
  tags        : string[];
  liveUrl     : string;
}

const ProductCard: React.FC<ProductCardProps> = ({
  title, price, platform, tags, liveUrl,
}) => {

  const [wishlisted, setWishlisted] = useState(false);
  const [hovered,   setHovered  ] = useState(false);

  return (
    <motion.article
      className    = "product-card"
      whileHover   = {{ y: -6, scale: 1.02 }}
      whileTap     = {{ scale: 0.98 }}
      transition   = {{ type: "spring", stiffness: 320, damping: 20 }}
      onHoverStart = {() => setHovered(true)}
      onHoverEnd   = {() => setHovered(false)}
    >
      <PlatformBadge  type    = {platform} />
      <h3             className = "card__title"  >{title}</h3>
      <p              className = "card__price"  >${price.toFixed(2)}</p>

      <ul className="card__tags">
        {tags.map((tag) => <TechTag key={tag} label={tag} />)}
      </ul>

      <footer className="card__actions">
        <Button href={liveUrl} variant="primary">View Product ↗</Button>

        <AnimatePresence>
          <motion.button
            key      = {String(wishlisted)}
            initial  = {{ scale: 0.8, opacity: 0 }}
            animate  = {{ scale: 1.0, opacity: 1 }}
            onClick  = {() => setWishlisted(w => !w)}
            aria-label = "Toggle wishlist"
          >
            {wishlisted ? "❤️" : "🤍"}
          </motion.button>
        </AnimatePresence>
      </footer>
    </motion.article>
  );
};

// ◀ composable · animated · accessible · type-safe ▶
export default ProductCard;
```
