## ⚛️ React Component Showcase

```tsx
// How I think in React — a reusable EcommerceCard from Skyrovix platform

import { useState } from "react";
import { motion } from "framer-motion";

interface ProductCardProps {
  title: string;
  price: number;
  platform: "etsy" | "ebay" | "pod" | "custom";
  tags: string[];
  liveUrl: string;
}

const ProductCard: React.FC<ProductCardProps> = ({
  title, price, platform, tags, liveUrl
}) => {
  const [wishlisted, setWishlisted] = useState(false);

  return (
    <motion.div
      className="product-card"
      whileHover={{ y: -4, scale: 1.02 }}
      transition={{ type: "spring", stiffness: 300 }}
    >
      <PlatformBadge type={platform} />

      <h3 className="product-card__title">{title}</h3>
      <p className="product-card__price">${price.toFixed(2)}</p>

      <div className="product-card__tags">
        {tags.map((tag) => <Tag key={tag} label={tag} />)}
      </div>

      <div className="product-card__actions">
        <Button href={liveUrl} variant="primary">View Product ↗</Button>
        <IconButton
          icon={wishlisted ? "heart-filled" : "heart"}
          onClick={() => setWishlisted(!wishlisted)}
          aria-label="Wishlist"
        />
      </div>
    </motion.div>
  );
};

// Every Skyrovix UI element: composable · animated · type-safe 🧩
```
